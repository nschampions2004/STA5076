\subsection*{Problem 5.28}

\Hypothesis
\begin{blockquote}
	$H_0: \ \mu = 35$ \\
	$H_{a}: \ \mu < 35$ 
\end{blockquote}

\Test Statistic
\begin{blockquote}
	$z_0 = \frac{\bar{y}-\mu_0}{\sigma/\sqrt{n}} = \frac{33.6-35}{9.2/\sqrt{40}} = -1.076$
\end{blockquote}

\Rejection Region
\begin{blockquote}
Reject if $z_0 \le z_\alpha$;$z_{0.01}=-2.575$
\end{blockquote}

\P-Value
\begin{blockquote}
With a confidence level of 99\% to compare to, my P-value ends up being 0.141.  
\end{blockquote}

\Conclusion / Interpretation
\begin{blockquote}
	We fail to reject the null hypothesis.  There is not sufficient evident to suggest that the number of improperly issued tickets is < 35.  
\end{blockquote}
